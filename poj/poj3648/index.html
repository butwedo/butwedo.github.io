一天之内完成两道2-sat，新方法就是霸气~

这道题的题目大意是一对新人要举行婚礼，邀请了n对夫妇在宴席上有这么些奇奇怪怪的风俗：一对夫妇不能做同一侧否则会有厄运；除此之外还有一些有暧昧关系的人，这些人不能同时被新娘看到，否则也会有厄运。问是否有解，若有的话输出任意一组。

这是题目链接。

以坐在新郎一侧的人是谁为对应点建图，记录新郎一侧的解，输出的时候反过来。不直接记录新娘一侧的点是因为据discuss说新娘也可能有暧昧关系。。。。。。然后还有一点就是要连一条新娘到新郎的边，表示新郎一定要选（因为是求新郎一侧的解）。

#include<iostream>
#include<stack> 
#include<cstdio>
#include<cstring>
using namespace std;
int tot;
int had[200],nxt[40004],point[40004];
inline void add(int x,int y)
{
    tot++;
    point[tot]=y;
    nxt[tot]=had[x];
    had[x]=tot;
}
int t,timel;
int dfn[200],low[200],blng[200];
bool vis[200];
stack<int>ss;
void tarjan(int x)
{
    dfn[x]=low[x]=t++;
    ss.push(x);
    vis[x]=1;
    for(int i=had[x];i;i=nxt[i])
    {
        int to=point[i];
        if(!dfn[to])
        {
            tarjan(to);
            low[x]=min(low[x],low[to]);
        }
        else
        if(vis[to])low[x]=min(low[x],dfn[to]);
    }
    if(dfn[x]==low[x])
    {
        timel++;
        while(!ss.empty())
        {
            int y=ss.top();
            blng[y]=timel;
            vis[y]=0;
            ss.pop();
            if(y==x)break;
        }
    }
}
int n,m;
int main()
{
    while(cin>>n>>m&&n)
    {    
        tot=0;
        t=0;
        timel=0;
        bool f=0;
        memset(dfn,0,sizeof(dfn));
        memset(low,0,sizeof(low));
        memset(had,0,sizeof(had));
        memset(nxt,0,sizeof(nxt));
        memset(blng,0,sizeof(blng));    
        memset(point,0,sizeof(point));
        for(int i=0;i<m;i++)
        {
            int w,h;
            char w1,h1;
            cin>>w>>w1;
            cin>>h>>h1;
            w<<=1;
            h<<=1;
            if(w1=='h')w++;
            if(h1=='h')h++;
            add(w,h^1);
            add(h,w^1);
        }
        add(0,1); 
        for(int i=0;i<2*n;i++)
            if(!dfn[i])tarjan(i);
        for(int i=0;i<2*n;i+=2)
            if(blng[i]==blng[i+1])
            {
                f=1;
                cout<<"bad luck\n";
                break;
            }
        if(f)continue;
        for(int i=2;i<2*n;i+=2)
        {
            cout<<(i>>1);
            if(blng[i]<blng[i+1])cout<<"h ";
            else cout<<"w ";
        }
        cout<<endl;
    }
    return 0;
}
